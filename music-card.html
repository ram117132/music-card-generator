let uploadedImage = '';

let chosenColor = '';


document.getElementById('imageInput').addEventListener('change', (e) => {

  const file = e.target.files[0];

  if (!file) return;

  const reader = new FileReader();

  reader.onload = (evt) => { uploadedImage = evt.target.result; };

  reader.readAsDataURL(file);

});


document.getElementById('colorPicker').addEventListener('input', function() {

  chosenColor = this.value;

});


function setPresetColor(color) {

  chosenColor = color;

  document.getElementById('colorPicker').value = color;

}


function autoPickColor() {

  if (!uploadedImage) { alert('Please upload an image first'); return; }

  const img = new Image();

  img.src = uploadedImage;

  img.crossOrigin = "Anonymous";

  img.onload = function() {

    const [r, g, b] = new ColorThief().getColor(img);

    const rgb = `rgb(${r}, ${g}, ${b})`;

    chosenColor = rgb;

    document.getElementById('cardPreview').style.backgroundColor = rgb;

    document.getElementById('cardPreview').style.display = 'block';

    document.getElementById('cardPreview').scrollIntoView({ behavior: 'smooth' });

  };

}

function downloadCard() {

  const card = document.getElementById('cardPreview');

  html2canvas(card, { useCORS: true, backgroundColor: null }).then(canvas => {

    const link = document.createElement('a');

    link.download = 'music-card.png';

    link.href = canvas.toDataURL('image/png');

    link.click();

  });

}

function seedBackground() {

  const layer = document.getElementById('bgLayer');

  const vw = window.innerWidth;

  const vh = window.innerHeight;

  const paths = ['driftA', 'driftB', 'driftC'];


  POSTERS.forEach(p => {

    const card = document.createElement('div');

    card.className = 'bg-card';

    card.style.width = `${110 + Math.random() * 40}px`;

    card.style.left = `${Math.random() * (vw - 140)}px`;

    card.style.top = `${Math.random() * (vh - 160)}px`;

    card.style.animationName = paths[Math.floor(Math.random() * paths.length)];

    card.style.animationDuration = `${9 + Math.random() * 10}s`;

    card.style.animationDelay = `${Math.random() * 6}s`;

    card.style.transform = `rotate(${(Math.random() * 6 - 3).toFixed(2)}deg)`;


    const img = document.createElement('img');

    img.src = p.img;

    img.alt = p.t;

    img.loading = 'eager';

    img.onerror = () => { img.src = 'https://via.placeholder.com/150x200?text=No+Image'; };

    card.appendChild(img);


    const meta = document.createElement('div');

    meta.className = 'meta';

    meta.innerHTML = `<div style="font-weight:700;font-size:12px;line-height:1.2">${p.t}</div>

                      <div style="font-size:11px;opacity:.9">${p.c}</div>`;

    card.appendChild(meta);


    layer.appendChild(card);

  });

}
